#include <stdio.h>

int main()
{
int TEAM1_runs=240;
int TEAM2_runs;
float overs_Done,left_overs,c,r;

    printf("team 1 scored=%d\n",TEAM1_runs);
    printf("enter the runs scored by the TEAM2_runs\n");
    scanf("%d",&TEAM2_runs);
    printf("enter the number of overs done\n");
    scanf("%f",&overs_Done);
    left_overs=50-overs_Done;

    r=(TEAM1_runs-TEAM2_runs)/left_overs;
    c=TEAM2_runs/overs_Done;
    printf("current run rate=%f\n",c);
    printf("required run rate=%f",r);
    return 0;
    
}
